```typescript
export interface PendingEffectData {
  kind: "burn-zone" | "smoke-zone";
  x: number;
  width: number;
  damage: number;
  turnsLeft: number;
}

export interface TankSnapshot { sessionId: string; x: number; hp: number; }
export interface BurnDamage   { sessionId: string; amount: number; }
export interface EffectResult {
  damages: BurnDamage[];
  survivors: PendingEffectData[];
}

export function processPendingEffects(
  effects: PendingEffectData[],
  tanks: TankSnapshot[],
): EffectResult {
  const damages: BurnDamage[] = [];
  const survivors: PendingEffectData[] = [];

  for (const effect of effects) {
    if (effect.kind === "burn-zone" && effect.damage > 0) {
      const half = effect.width / 2;
      for (const tank of tanks) {
        if (Math.abs(tank.x - effect.x) <= half) {
          damages.push({ sessionId: tank.sessionId, amount: effect.damage });
        }
      }
    }
    const next: PendingEffectData = { ...effect, turnsLeft: effect.turnsLeft - 1 };
    if (next.turnsLeft > 0) survivors.push(next);
  }

  return { damages, survivors };
}
```

// In the onTurnReady callback / wherever the turn starts:
private runPendingEffects(): void {
  const { state, terrain } = this;
  if (state.pendingEffects.length === 0) return;

  const tankSnapshots = Array.from(state.tanks.values())
    .filter(t => t.alive)
    .map(t => ({ sessionId: t.sessionId, x: t.x, hp: t.hp }));

  const effectData = state.pendingEffects.map(e => ({
    kind: e.kind as "burn-zone" | "smoke-zone",
    x: e.x, width: e.width, damage: e.damage, turnsLeft: e.turnsLeft,
  }));

  const { damages, survivors } = processPendingEffects(effectData, tankSnapshots);

  // Apply burn damages
  for (const d of damages) {
    const tank = state.tanks.get(d.sessionId);
    if (tank?.alive) {
      tank.hp = Math.max(0, tank.hp - d.amount);
      if (tank.hp <= 0) tank.alive = false;
    }
  }
  if (damages.length > 0) {
    this.broadcast("burn-tick", { damages });
  }

  // Replace effect array with survivors
  state.pendingEffects.clear();
  for (const s of survivors) {
    const e = new PendingEffect();
    e.kind = s.kind; e.x = s.x; e.width = s.width; e.damage = s.damage; e.turnsLeft = s.turnsLeft;
    state.pendingEffects.push(e);
  }
}

```typescript
import type { TargetInfo, DamageEntry } from "../types";

const HIT_THRESHOLD = 18; // px — how close to beam center counts as a hit

export interface LaserInput {
  originX: number;
  originY: number;
  angleDeg: number;      // 0=left, 90=up, 180=right (standard game angle)
  targets: TargetInfo[];
  damage: number;
  terrain: Int16Array;
  terrainWidth: number;
  terrainHeight: number;
}

export interface LaserResult {
  endX: number;
  endY: number;
  damages: DamageEntry[];
}

export function resolveLaserBeam(input: LaserInput): LaserResult {
  const { originX, originY, angleDeg, targets, damage, terrain, terrainWidth, terrainHeight } = input;

  // Direction vector (same convention as game: angle 180=right, 0=left)
  const rad = (angleDeg * Math.PI) / 180;
  const dx = -Math.cos(rad); // negative because angle 0 = left
  const dy = -Math.sin(rad);

  let x = originX;
  let y = originY;
  const step = 2; // px per step
  const maxDist = Math.sqrt(terrainWidth ** 2 + terrainHeight ** 2);

  const hitTanks = new Set<string>();

  for (let dist = 0; dist < maxDist; dist += step) {
    x = originX + dx * dist;
    y = originY + dy * dist;

    // Out of bounds
    if (x < 0 || x >= terrainWidth || y < 0 || y >= terrainHeight) break;

    // Terrain hit
    const surfaceY = terrain[Math.floor(x)] ?? terrainHeight;
    if (y >= surfaceY) break;

    // Check tanks
    for (const t of targets) {
      if (hitTanks.has(t.playerId)) continue;
      const perpDist = Math.abs((t.x - x) * dy - (t.y - y) * dx);
      const alongDist = (t.x - originX) * dx + (t.y - originY) * dy;
      if (perpDist < HIT_THRESHOLD && alongDist >= 0 && alongDist <= dist + step) {
        hitTanks.add(t.playerId);
      }
    }
  }

  const damages: DamageEntry[] = [];
  for (const t of targets) {
    if (!hitTanks.has(t.playerId)) continue;
    const shieldAbsorbed = Math.min(damage, t.shieldHp);
    const hullDamage = damage - shieldAbsorbed;
    damages.push({ playerId: t.playerId, amount: damage, shieldDamage: shieldAbsorbed, hullDamage });
  }

  return { endX: x, endY: y, damages };
}
```

updateZones(zones: Array<{ kind: "burn-zone" | "smoke-zone"; x: number; width: number }>): void {
  this.zoneOverlay.clear();
  const now = Date.now();
  for (const zone of zones) {
    const left  = zone.x - zone.width / 2;
    const right = zone.x + zone.width / 2;
    if (zone.kind === "burn-zone") {
      // Flickering orange flame strip at terrain surface
      const alpha = 0.4 + 0.2 * Math.sin(now / 120); // flicker
      this.zoneOverlay.rect(left, 0, zone.width, 1600)
        .fill({ color: 0xff6600, alpha });
    } else {
      // Semi-transparent gray smoke cloud
      this.zoneOverlay.rect(left, 0, zone.width, 1600)
        .fill({ color: 0x888888, alpha: 0.35 });
    }
  }
}
